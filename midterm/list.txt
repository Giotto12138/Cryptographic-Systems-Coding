sort(list,people,num)
{
    for i=0; i<num-1; i++
    {
        for j=i+1; j<num; j++
        {
            if list[i]>list[j]  // if the value is larger than its next value, exchange them
            {
                temp = list[i]
                list[i] = list[j]
                list[j] = temp

                temp = people[i]    // exchange people at the same time
                people[i] = people[j]
                people[j] = temp
            }
        }
    }

    return list
}
